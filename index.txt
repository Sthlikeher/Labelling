            <!-- Messages will be added here -->
        </div>

            <div class="input-wrapper">
                <input 
                    type="text" 
                    id="messageInput" 
                    placeholder="Nhập câu hỏi của bạn..."
                >
            </div>
            <button class="send-btn" id="sendBtn">Gửi</button>
        </div>
    </div>

    <script>
        const messagesArea = document.getElementById('messagesArea');
        const messageInput = document.getElementById('messageInput');
        const sendBtn = document.getElementById('sendBtn');

        // Initial message
        const initialMessages = [
            {
                type: 'user',
                text: 'Bạn: Làm thế nào để đối phó với định kiến?'
            },
            {
                type: 'system',
                text: 'Hệ thống LLM: Chào bạn. Việc đối phó với định kiến cần sự kiên nhẫn. Hãy tập trung vào giá trị và mục tiêu của bạn thay vì có gắng thay đổi ý kiến của mọi người.'
            }
        ];

        function addMessage(type, text) {
            const messageDiv = document.createElement('div');
            messageDiv.className = `message ${type === 'user' ? 'user-message' : 'system-message'}`;
            
            if (type === 'user') {
                messageDiv.innerHTML = `
                    <div class="user-label">Bạn</div>
                    <div class="user-text">${text}</div>
                `;
            } else {
                messageDiv.innerHTML = `
                    <div class="system-label">Hệ thống LLM</div>
                    <div class="system-text">${text}</div>
                `;
            }
            
            messagesArea.appendChild(messageDiv);
            messagesArea.scrollTop = messagesArea.scrollHeight;
        }

        // Load initial messages
        initialMessages.forEach(msg => {
            addMessage(msg.type, msg.text);
        });

        // Send message function
        function sendMessage() {
            const text = messageInput.value.trim();
            if (!text) return;

            addMessage('user', text);
            messageInput.value = '';

            // Simulate system response
            setTimeout(() => {
                const responses = [
                    'Cảm ơn bạn đã chia sẻ. Đó là một điểm rất tốt để xem xét.',
                    'Tôi hiểu ý của bạn. Hãy tiếp tục phát triển ý tưởng này.',
                    'Đó là một câu hỏi thú vị. Chúng ta có thể tìm hiểu sâu hơn về vấn đề này.',
                    'Bạn đã đặt câu hỏi tuyệt vời. Hãy để tôi giúp bạn tìm ra câu trả lời.'
                ];
                const randomResponse = responses[Math.floor(Math.random() * responses.length)];
                addMessage('system', randomResponse);
            }, 600);
        }

        // Event listeners
        sendBtn.addEventListener('click', sendMessage);
        messageInput.addEventListener('keypress', (e) => {
            if (e.key === 'Enter') {
                sendMessage();
            }
        });

        // Focus input on load
        messageInput.focus();
    </script>
